⚙ Connect student data sheet

How it works: You paste a small script into your Google Sheet's Apps Script editor (Extensions → Apps Script). Deploy it once as a web app — no Google Cloud account or sign-in needed. Paste the URL below.
📋 Step 1 — Paste this script into your sheet's Apps Script editor 
function doGet(e) {
  const action = e.parameter.action || 'read';
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  // ── WRITE: receive a chunk of ISP data ──────────────────────────
  if (action === 'chunk') {
    const props = PropertiesService.getScriptProperties();
    const id = e.parameter.id;
    const n  = e.parameter.n;
    props.setProperty('chunk_' + id + '_' + n, e.parameter.data);
    props.setProperty('chunk_' + id + '_total', e.parameter.total);
    return json({ ok: true });
  }

  // ── WRITE: reassemble chunks and save ISP ───────────────────────
  if (action === 'commit') {
    const props = PropertiesService.getScriptProperties();
    const id    = e.parameter.id;
    const total = parseInt(props.getProperty('chunk_' + id + '_total') || '0');
    let assembled = '';
    for (let i = 0; i < total; i++) {
      assembled += props.getProperty('chunk_' + id + '_' + i) || '';
    }
    // Clean up temp keys
    for (let i = 0; i < total; i++) props.deleteProperty('chunk_' + id + '_' + i);
    props.deleteProperty('chunk_' + id + '_total');

    const isp = JSON.parse(assembled);

    let ispSheet = ss.getSheetByName('ISP_Data');
    if (!ispSheet) {
      ispSheet = ss.insertSheet('ISP_Data');
      ispSheet.getRange(1, 1, 1, 6).setValues([['id','arborId','name','level','updated','data']]);
    }
    const rows = ispSheet.getDataRange().getValues();
    const idCol = rows[0].indexOf('id');
    let rowIdx = -1;
    for (let i = 1; i < rows.length; i++) {
      if (String(rows[i][idCol]) === String(isp.id)) { rowIdx = i + 1; break; }
    }
    const rowData = [isp.id, isp.arborId||'', isp.name||'', isp.level||'', isp.updated||'', JSON.stringify(isp)];
    if (rowIdx === -1) ispSheet.appendRow(rowData);
    else ispSheet.getRange(rowIdx, 1, 1, rowData.length).setValues([rowData]);

    return json({ ok: true, id: isp.id });
  }

  // ── READ: return students + all saved ISPs ──
  const studentSheet = ss.getSheets()[0];
  const [headers, ...rows] = studentSheet.getDataRange().getValues();
  const students = rows.map(row => {
    const obj = {};
    headers.forEach((h, i) => { if (h) obj[String(h).trim()] = row[i]; });
    return obj;
  }).filter(s => s['Arbor ID']);

  let isps = [];
  const ispSheet = ss.getSheetByName('ISP_Data');
  if (ispSheet && ispSheet.getLastRow() > 1) {
    const ispRows = ispSheet.getDataRange().getValues();
    const dataCol = ispRows[0].indexOf('data');
    if (dataCol > -1) {
      isps = ispRows.slice(1)
        .map(r => { try { return JSON.parse(r[dataCol]); } catch(e2) { return null; } })
        .filter(Boolean);
    }
  }

  return json({ students, isps });
}

function doPost(e) { return doGet(e); }

function json(data) {
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
🚀 Step 2 — Deploy the script as a web app
  1. In the Apps Script editor, click Deploy → New deployment
  2. Click the gear icon next to "Type" and choose Web app
  3. Set Execute as: Me
  4. Set Who has access: Anyone
  5. Click Deploy and copy the web app URL
Privacy note: Student data is fetched directly from your sheet into this browser session only. Nothing is sent to any third-party server. ISP data is written back only to your own sheet via the script you control.
